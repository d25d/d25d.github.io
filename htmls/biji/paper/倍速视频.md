// 核心倍速函数
const buttons = document.querySelectorAll('#speedButtons button');
function setActive(speed) {
    buttons.forEach(btn => btn.classList.remove('active'));
}
function s(v) {
    const speed = parseFloat(v);
    if (isNaN(speed) || speed < 0.1 || speed > 16) return;
    document.querySelectorAll('video').forEach(i => i.playbackRate = speed);
    setActive(); // 清空所有高亮
}

// 绑定按钮
buttons.forEach(btn => {
    btn.addEventListener('click', () => {
        s(btn.dataset.speed);
        buttons.forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
    });
});

// 绑定自定义倍速（修复完成）
document.getElementById('customBtn').addEventListener('click', () => {
    const speed = document.getElementById('cs').value.trim();
    s(speed);
});

// 回车也能触发
document.getElementById('cs').addEventListener('keyup', e => {
    if(e.key === 'Enter') s(e.target.value);
});
